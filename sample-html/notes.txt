            function filterArtworks() {
                const nameFilter = $('#artworkName').val().toLowerCase();
                const artistFilter = $('#artistName').val().toLowerCase();
                const typeFilter = $('#artworkType').val().toLowerCase();

                $('.artwork-card').each(function() {
                    const name = $(this).data('name').toLowerCase();
                    const artist = $(this).data('artist').toLowerCase();
                    const type = $(this).data('type').toLowerCase();

                    if ((name.includes(nameFilter) || !nameFilter) &&
                        (artist.includes(artistFilter) || !artistFilter) &&
                        (type.includes(typeFilter) || !typeFilter)) {
                        $(this).show();
                    } else {
                        $(this).hide();
                    }
                });
            }
